\begin{haskell}
data Tree a = Leaf a | Bin (Tree a) a (Tree a)

sumTree :: Tree Int -> Int
sumTree (Leaf x)    = x
sumTree (Bin l x r) = x + sumTree l + sumTree r

exampleTree = Bin (Bin (Leaf 1) 3 (Leaf 2)) 5 (Bin (Leaf 1) 3 (Leaf 2))
\end{haskell}
